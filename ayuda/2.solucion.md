# Contador de palabras
```java
    public static void main() {
        // Creamos el objeto Scanner para leer de teclado
        Scanner sc = new Scanner(System.in);

        // Pedimos al usuario la frase
        System.out.println("Introduce una frase");
        String listaCompra = sc.nextLine();

        // Iniciamos el numero de palabras a 0 y la posicion a partir de la que miramos (0)
        int posicion = 0;
        int contadorPalabras = 0;

        // Aumentamos el numero de palabras contadas y, tras ello, actualizamos posicion hasta el próximo espacio.
        // Repetiremos la posicion hasta que no hayan mas espacios (lista.indexOf(" ", posicion + 1) devolverá -1 si no hay mas espacios)
        do {
            contadorPalabras++;
            posicion = listaCompra.indexOf(" ", posicion + 1);
        } while (posicion > 0);

        // indicamos el numero de palabras
        System.out.println("Tu frase tiene " + contadorPalabras + " palabras");
    }
```

```java
            String objetoListaCompra = listaCompra.substring(posicion, posicionProximaComa);

            // Localizamos la proxima coma que haya a partir de nuestra posicion actual actualizando la variable de prixima cima
            posicionProximaComa = listaCompra.indexOf(",", posicion);
        }
    }
```

## Clase Almacen
```java
package org.example;

public class Almacen {
    // Creamos los atributos de latas en stock y los inicializamos a 0
    private int classicInStock = 0;
    private int negroLocoInStock = 0;
    private int pinkExplosionInStock = 0;

    // Creamos el atributo de capacidad total sin inicializar
    private int totalCapacity;

    // Definimos el contructor de almacen
    public Almacen(int totalCapacity) {
        // Inicializamos el atributo `totalCapacity`
        this.totalCapacity = totalCapacity;
    }

    // definimos la funcion privada de countLatas()
    private int countLatas() {
        // devuelve la suma del stock de los tres tipos de latas
        return classicInStock + negroLocoInStock + pinkExplosionInStock;
    }

    // definimos la funcion publica de addLata que devolvera true o false
    public boolean addLata(String tipoLata, int numeroLatas) {
        // miramos si podemos añadir el numero de latas
        if (this.countLatas()  + numeroLatas > this.totalCapacity) {
            // sino, devolvemos false
            return false;
        }
        // miramos el tipo de lata indicado por el usuario
        switch (tipoLata) {
            // si es classic
            case "classic":
                // aumentamos el stock de classic
                this.classicInStock += numeroLatas;
                break;
            // si es negroloco
            case "negroloco":
                // aumentamos el stock de negroloco
                this.negroLocoInStock += numeroLatas;
                break;
            // si es pinkexplosion
            case "pinkexplosion":
                // aumentamos el stock de pinkExplosion
                this.pinkExplosionInStock += numeroLatas;
                break;
        }
        // devolvemos true
        return true;
    }
    // definimos la funcion de warehouse status
    public void warehouseStatus() {
        // mostramos por pantalla los distintos atributos del almacen de stock y capacidad
        System.out.println(" === Status del Almacen ===");
        System.out.println(" Classic: " + this.classicInStock );
        System.out.println(" Negro Loco: " + this.negroLocoInStock );
        System.out.println(" Pink Explosion: " + this.pinkExplosionInStock );
        System.out.println(" Espacio disponible: " + (this.totalCapacity - this.countLatas()) );
        System.out.println(" Uso actual: " + (this.countLatas() * 100)/this.totalCapacity + "%");
        System.out.println(" ======================== ");
        System.out.println();

    }
}
```

```java
package org.example;

import java.util.Scanner;

public class Programa {


    public static void main(String[] args) {
        // Creamos el objeto Scanner para leer de teclado
        Scanner input = new Scanner(System.in);
        // Preguntamos al usuario el tamaño total del almacen y lo leemos
        System.out.println("Indica el tamaño total del almacen: ");
        int number = input.nextInt();
        // Creamos un objeto de tipo almacen con el tamaño de almacen que nos ha indicado el usuario
        Almacen almacen = new Almacen(number);
        
        // definimos la variable que indicara la opcion del usuario
        int opcion;
        do {
            // mostramos al usuario el menu
            System.out.println("Indica que quieres hacer: ");
            System.out.println("0 - Salir ");
            System.out.println("1 - Añadir latas");
            System.out.println("2 - Revisar el estado");
            System.out.println();
            // esperamos a que el usuario indique la accion
            opcion  = input.nextInt();
            
            // en funcion de la accion
            switch (opcion) {
                // si es 1 (añadir latas)
                case 1:
                    // pedimos al usuario que diga el tipo de lata y cuantas latas separado por coma
                    System.out.println("Indica el tipo de lata y la cantida a almacenar: ");
                    String informacion = input.next();
                    System.out.println();
                    
                    // buscamos la posicion de la coma (",")
                    int posicionComa = informacion.indexOf(",");

                    // Extraemos el tipo, sabiendo que sera la palabra que este entre la posicon 0 y la coma
                    String tipo = informacion.substring(0, posicionComa);
                    
                    // extraemos la cantidad y la parseamos a entero, sabiendo que sea lo que haya despues de la coma
                    int cantidad =  Integer.parseInt(informacion.substring(posicionComa + 1));

                    // llamamos a la funcion de añadir latas y revisamos su resultado
                    boolean sePudoAnadir = almacen.addLata(tipo, cantidad);
                    // si se pudieron añadir
                    if (sePudoAnadir)
                        // lo indicamos
                        System.out.println("Latas añadidas correctamentet");
                        // si NO se pudieron añadir
                    else
                        // lo indicamos
                        System.out.println("No se han podido añadir: almacen lleno");
                    System.out.println();
                    break;
                // si el usuario indica la opcion 2 (show)
                case 2:
                    // ejecutamos la funcion warehouseStatus, que mostrara por pantalla la informacion.
                    almacen.warehouseStatus();
                    break;
            }
        // esto lo haremos mientras el usuario no elija la opcion 0 (salir)
        } while(opcion != 0);

    }
}
```
